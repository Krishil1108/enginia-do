import React, { useState, useEffect } from 'react';
import { Calendar, Users, Bell, MessageCircle, Mic, Plus, Edit2, Trash2, MoreVertical, Filter, Check, Clock, AlertCircle, X, ChevronLeft, ChevronRight } from 'lucide-react';

const PRIORITY_COLORS = {
  Low: 'bg-green-100 text-green-800 border-green-300',
  Medium: 'bg-yellow-100 text-yellow-800 border-yellow-300',
  High: 'bg-red-100 text-red-800 border-red-300'
};

const SEVERITY_BADGES = {
  Minor: 'bg-blue-100 text-blue-700',
  Major: 'bg-orange-100 text-orange-700',
  Critical: 'bg-red-100 text-red-700'
};

const TEAMS = ['Team A', 'Team B', 'Team C'];
const PROJECTS = ['Website Redesign', 'Mobile App', 'Marketing Campaign', 'Infrastructure'];

const TaskManagementSystem = () => {
  const [currentView, setCurrentView] = useState('my tasks');
  const [tasks, setTasks] = useState([]);
  const [showTaskModal, setShowTaskModal] = useState(false);
  const [selectedProject, setSelectedProject] = useState(null);
  const [editingTask, setEditingTask] = useState(null);
  const [filters, setFilters] = useState({});
  const [showMenu, setShowMenu] = useState(null);
  const [isRecording, setIsRecording] = useState(false);
  const [currentUser, setCurrentUser] = useState('John Doe'); // Default user
  const [showUserSelector, setShowUserSelector] = useState(false);
  const [showAdvancedMenu, setShowAdvancedMenu] = useState(false);

  // Load tasks from storage
  useEffect(() => {
    loadTasks();
  }, []);

  const loadTasks = async () => {
    try {
      const result = await window.storage.get('tasks');
      if (result && result.value) {
        setTasks(JSON.parse(result.value));
      }
    } catch (error) {
      console.log('No existing tasks found');
    }
  };

  const saveTasks = async (updatedTasks) => {
    try {
      await window.storage.set('tasks', JSON.stringify(updatedTasks));
      setTasks(updatedTasks);
    } catch (error) {
      console.error('Error saving tasks:', error);
    }
  };

  const [formData, setFormData] = useState({
    project: '',
    title: '',
    description: '',
    priority: 'Medium',
    severity: 'Minor',
    inDate: '',
    outDate: '',
    team: 'Team A',
    associates: [],
    reminder: '',
    whatsapp: false,
    status: 'Pending'
  });

  const resetForm = () => {
    setFormData({
      project: selectedProject || '',
      title: '',
      description: '',
      priority: 'Medium',
      severity: 'Minor',
      inDate: '',
      outDate: '',
      team: 'Team A',
      associates: [],
      reminder: '',
      whatsapp: false,
      status: 'Pending'
    });
    setEditingTask(null);
  };

  const handleSubmit = async () => {
    if (!formData.project || !formData.title || !formData.inDate || !formData.outDate) {
      alert('Please fill in all required fields');
      return;
    }
    
    const newTask = {
      ...formData,
      id: editingTask ? editingTask.id : Date.now(),
      createdAt: editingTask ? editingTask.createdAt : new Date().toISOString()
    };

    const updatedTasks = editingTask 
      ? tasks.map(t => t.id === editingTask.id ? newTask : t)
      : [...tasks, newTask];

    await saveTasks(updatedTasks);
    setShowTaskModal(false);
    resetForm();
  };

  const deleteTask = async (id) => {
    const updatedTasks = tasks.filter(t => t.id !== id);
    await saveTasks(updatedTasks);
  };

  const editTask = (task) => {
    setFormData(task);
    setEditingTask(task);
    setShowTaskModal(true);
  };

  const getFilteredTasks = () => {
    return tasks.filter(task => {
      if (filters.project && task.project !== filters.project) return false;
      if (filters.team && task.team !== filters.team) return false;
      if (filters.priority && task.priority !== filters.priority) return false;
      if (filters.severity && task.severity !== filters.severity) return false;
      return true;
    });
  };

  const getTaskStats = () => {
    const filtered = getFilteredTasks();
    return {
      total: filtered.length,
      pending: filtered.filter(t => t.status === 'Pending').length,
      completed: filtered.filter(t => t.status === 'Completed').length,
      overdue: filtered.filter(t => {
        if (t.status === 'Completed') return false;
        return new Date(t.outDate) < new Date();
      }).length
    };
  };

  const voiceInput = () => {
    setIsRecording(!isRecording);
    // Placeholder for voice input functionality
    setTimeout(() => setIsRecording(false), 2000);
  };

  // Dashboard View
  const DashboardView = () => {
    const stats = getTaskStats();
    
    return (
      <div className="space-y-6">
        <div className="grid grid-cols-1 md:grid-cols-4 gap-4">
          <StatCard title="Total Tasks" value={stats.total} icon={<Calendar />} color="blue" />
          <StatCard title="Pending" value={stats.pending} icon={<Clock />} color="yellow" />
          <StatCard title="Completed" value={stats.completed} icon={<Check />} color="green" />
          <StatCard title="Overdue" value={stats.overdue} icon={<AlertCircle />} color="red" />
        </div>

        <div className="bg-white rounded-xl p-6 shadow-sm border border-gray-100">
          <h3 className="text-lg font-semibold mb-4 flex items-center gap-2">
            <Filter className="w-5 h-5" />
            Quick Filters
          </h3>
          <div className="grid grid-cols-1 md:grid-cols-5 gap-4">
            <FilterDropdown label="Project" options={PROJECTS} value={filters.project} onChange={(v) => setFilters({...filters, project: v})} />
            <FilterDropdown label="Team" options={TEAMS} value={filters.team} onChange={(v) => setFilters({...filters, team: v})} />
            <FilterDropdown label="Priority" options={['Low', 'Medium', 'High']} value={filters.priority} onChange={(v) => setFilters({...filters, priority: v})} />
            <FilterDropdown label="Severity" options={['Minor', 'Major', 'Critical']} value={filters.severity} onChange={(v) => setFilters({...filters, severity: v})} />
            <button 
              onClick={() => setFilters({})}
              className="px-4 py-2 text-sm text-gray-600 hover:text-gray-900 border border-gray-200 rounded-lg hover:bg-gray-50 transition-colors"
            >
              Clear All
            </button>
          </div>
        </div>

        <ProjectTasksView />
      </div>
    );
  };

  // Project-wise Task List View
  const ProjectTasksView = () => {
    const groupedTasks = getFilteredTasks().reduce((acc, task) => {
      if (!acc[task.project]) acc[task.project] = [];
      acc[task.project].push(task);
      return acc;
    }, {});

    return (
      <div className="space-y-6">
        {Object.entries(groupedTasks).map(([project, projectTasks]) => (
          <div key={project} className="bg-white rounded-xl shadow-sm border border-gray-100 overflow-hidden">
            <div className="bg-gray-50 px-6 py-4 border-b border-gray-100 flex justify-between items-center">
              <h3 className="font-semibold text-gray-900">{project}</h3>
              <button
                onClick={() => {
                  setSelectedProject(project);
                  setFormData({...formData, project});
                  setShowTaskModal(true);
                }}
                className="flex items-center gap-2 px-4 py-2 bg-blue-600 text-white rounded-lg hover:bg-blue-700 transition-colors text-sm"
              >
                <Plus className="w-4 h-4" />
                Add Task
              </button>
            </div>
            <div className="divide-y divide-gray-100">
              {projectTasks.map(task => (
                <TaskRow key={task.id} task={task} onEdit={editTask} onDelete={deleteTask} />
              ))}
            </div>
          </div>
        ))}
        {Object.keys(groupedTasks).length === 0 && (
          <div className="text-center py-12 bg-white rounded-xl border border-gray-100">
            <p className="text-gray-500">No tasks found. Create your first task!</p>
          </div>
        )}
      </div>
    );
  };

  // Timeline/Gantt View
  const TimelineView = () => {
    const filtered = getFilteredTasks().filter(t => t.inDate && t.outDate);
    const dates = filtered.flatMap(t => [new Date(t.inDate), new Date(t.outDate)]);
    if (dates.length === 0) {
      return (
        <div className="bg-white rounded-xl p-12 text-center border border-gray-100">
          <Calendar className="w-12 h-12 mx-auto text-gray-300 mb-4" />
          <p className="text-gray-500">No tasks with dates to display in timeline</p>
        </div>
      );
    }

    const minDate = new Date(Math.min(...dates));
    const maxDate = new Date(Math.max(...dates));
    const daysDiff = Math.ceil((maxDate - minDate) / (1000 * 60 * 60 * 24)) + 1;

    return (
      <div className="bg-white rounded-xl p-6 shadow-sm border border-gray-100 overflow-x-auto">
        <h3 className="text-lg font-semibold mb-6">Timeline View</h3>
        <div className="min-w-max">
          <div className="flex gap-2 mb-4 text-xs text-gray-500">
            {Array.from({length: Math.min(daysDiff, 30)}, (_, i) => {
              const date = new Date(minDate);
              date.setDate(date.getDate() + i);
              return (
                <div key={i} className="w-24 text-center">
                  {date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' })}
                </div>
              );
            })}
          </div>
          <div className="space-y-3">
            {filtered.map(task => {
              const start = new Date(task.inDate);
              const end = new Date(task.outDate);
              const startDay = Math.ceil((start - minDate) / (1000 * 60 * 60 * 24));
              const duration = Math.ceil((end - start) / (1000 * 60 * 60 * 24)) + 1;
              
              return (
                <div key={task.id} className="relative flex items-center">
                  <div className="w-48 pr-4 text-sm font-medium truncate">{task.title}</div>
                  <div className="flex-1 relative h-8">
                    <div 
                      className={`absolute h-full rounded-lg ${task.priority === 'High' ? 'bg-red-400' : task.priority === 'Medium' ? 'bg-yellow-400' : 'bg-green-400'} hover:opacity-80 transition-opacity cursor-pointer flex items-center px-3 text-xs font-medium text-white`}
                      style={{
                        left: `${startDay * 6}rem`,
                        width: `${duration * 6}rem`
                      }}
                      title={`${task.title} (${task.priority})`}
                    >
                      {task.title}
                    </div>
                  </div>
                </div>
              );
            })}
          </div>
        </div>
      </div>
    );
  };

  // Associate View
  const AssociateView = () => {
    const associateTasks = {};
    getFilteredTasks().forEach(task => {
      task.associates.forEach(associate => {
        if (!associateTasks[associate]) associateTasks[associate] = [];
        associateTasks[associate].push(task);
      });
    });

    return (
      <div className="space-y-6">
        {Object.entries(associateTasks).map(([associate, tasks]) => (
          <div key={associate} className="bg-white rounded-xl shadow-sm border border-gray-100 p-6">
            <h3 className="font-semibold text-lg mb-4 flex items-center gap-2">
              <Users className="w-5 h-5" />
              {associate}
              <span className="ml-auto text-sm text-gray-500">{tasks.length} tasks</span>
            </h3>
            <div className="space-y-3">
              {tasks.map(task => (
                <div key={task.id} className="flex items-center justify-between p-4 bg-gray-50 rounded-lg">
                  <div className="flex-1">
                    <div className="font-medium">{task.title}</div>
                    <div className="text-sm text-gray-500 mt-1">{task.project}</div>
                  </div>
                  <div className="flex items-center gap-3">
                    <span className={`px-3 py-1 rounded-full text-xs font-medium ${PRIORITY_COLORS[task.priority]}`}>
                      {task.priority}
                    </span>
                    {task.whatsapp && (
                      <MessageCircle className="w-5 h-5 text-green-600" />
                    )}
                  </div>
                </div>
              ))}
            </div>
          </div>
        ))}
        {Object.keys(associateTasks).length === 0 && (
          <div className="text-center py-12 bg-white rounded-xl border border-gray-100">
            <Users className="w-12 h-12 mx-auto text-gray-300 mb-4" />
            <p className="text-gray-500">No associates assigned to tasks yet</p>
          </div>
        )}
      </div>
    );
  };

  // User View - Individual user's task view
  const UserView = () => {
    const userTasks = getFilteredTasks().filter(task => 
      task.associates.includes(currentUser)
    );

    const todayTasks = userTasks.filter(task => {
      const today = new Date().toDateString();
      const dueDate = new Date(task.outDate).toDateString();
      return dueDate === today && task.status !== 'Completed';
    });

    const upcomingTasks = userTasks.filter(task => {
      const now = new Date();
      const dueDate = new Date(task.outDate);
      return dueDate > now && task.status !== 'Completed';
    }).sort((a, b) => new Date(a.outDate) - new Date(b.outDate));

    const overdueTasks = userTasks.filter(task => {
      const now = new Date();
      const dueDate = new Date(task.outDate);
      return dueDate < now && task.status !== 'Completed';
    });

    const completedTasks = userTasks.filter(task => task.status === 'Completed');

    const toggleTaskStatus = async (taskId) => {
      const updatedTasks = tasks.map(t => {
        if (t.id === taskId) {
          return { ...t, status: t.status === 'Completed' ? 'Pending' : 'Completed' };
        }
        return t;
      });
      await saveTasks(updatedTasks);
    };

    return (
      <div className="space-y-6">
        {/* User Stats */}
        <div className="bg-gradient-to-r from-blue-500 to-blue-600 rounded-xl p-6 text-white shadow-lg">
          <div className="flex items-center justify-between mb-4">
            <div>
              <h2 className="text-2xl font-bold">Welcome, {currentUser}!</h2>
              <p className="text-blue-100 mt-1">Here's your task overview</p>
            </div>
            <button
              onClick={() => setShowUserSelector(!showUserSelector)}
              className="px-4 py-2 bg-white bg-opacity-20 hover:bg-opacity-30 rounded-lg transition-colors text-sm font-medium"
            >
              Switch User
            </button>
          </div>
          <div className="grid grid-cols-2 md:grid-cols-4 gap-4">
            <div className="bg-white bg-opacity-20 rounded-lg p-4">
              <div className="text-3xl font-bold">{userTasks.length}</div>
              <div className="text-blue-100 text-sm mt-1">Total Tasks</div>
            </div>
            <div className="bg-white bg-opacity-20 rounded-lg p-4">
              <div className="text-3xl font-bold">{todayTasks.length}</div>
              <div className="text-blue-100 text-sm mt-1">Due Today</div>
            </div>
            <div className="bg-white bg-opacity-20 rounded-lg p-4">
              <div className="text-3xl font-bold">{overdueTasks.length}</div>
              <div className="text-blue-100 text-sm mt-1">Overdue</div>
            </div>
            <div className="bg-white bg-opacity-20 rounded-lg p-4">
              <div className="text-3xl font-bold">{completedTasks.length}</div>
              <div className="text-blue-100 text-sm mt-1">Completed</div>
            </div>
          </div>
        </div>

        {/* User Selector Dropdown */}
        {showUserSelector && (
          <div className="bg-white rounded-xl shadow-sm border border-gray-100 p-4">
            <h3 className="font-semibold mb-3">Select User</h3>
            <div className="grid grid-cols-2 md:grid-cols-4 gap-2">
              {Array.from(new Set(tasks.flatMap(t => t.associates))).map(user => (
                <button
                  key={user}
                  onClick={() => {
                    setCurrentUser(user);
                    setShowUserSelector(false);
                  }}
                  className={`px-4 py-2 rounded-lg text-sm font-medium transition-colors ${
                    currentUser === user
                      ? 'bg-blue-600 text-white'
                      : 'bg-gray-100 text-gray-700 hover:bg-gray-200'
                  }`}
                >
                  {user}
                </button>
              ))}
            </div>
          </div>
        )}

        {/* Overdue Tasks */}
        {overdueTasks.length > 0 && (
          <div className="bg-white rounded-xl shadow-sm border border-red-200">
            <div className="bg-red-50 px-6 py-4 border-b border-red-100 flex items-center gap-2">
              <AlertCircle className="w-5 h-5 text-red-600" />
              <h3 className="font-semibold text-red-900">Overdue Tasks ({overdueTasks.length})</h3>
            </div>
            <div className="divide-y divide-gray-100">
              {overdueTasks.map(task => (
                <UserTaskCard key={task.id} task={task} onToggleStatus={toggleTaskStatus} />
              ))}
            </div>
          </div>
        )}

        {/* Due Today */}
        {todayTasks.length > 0 && (
          <div className="bg-white rounded-xl shadow-sm border border-orange-200">
            <div className="bg-orange-50 px-6 py-4 border-b border-orange-100 flex items-center gap-2">
              <Clock className="w-5 h-5 text-orange-600" />
              <h3 className="font-semibold text-orange-900">Due Today ({todayTasks.length})</h3>
            </div>
            <div className="divide-y divide-gray-100">
              {todayTasks.map(task => (
                <UserTaskCard key={task.id} task={task} onToggleStatus={toggleTaskStatus} />
              ))}
            </div>
          </div>
        )}

        {/* Upcoming Tasks */}
        {upcomingTasks.length > 0 && (
          <div className="bg-white rounded-xl shadow-sm border border-gray-100">
            <div className="bg-gray-50 px-6 py-4 border-b border-gray-100 flex items-center gap-2">
              <Calendar className="w-5 h-5 text-gray-600" />
              <h3 className="font-semibold text-gray-900">Upcoming Tasks ({upcomingTasks.length})</h3>
            </div>
            <div className="divide-y divide-gray-100">
              {upcomingTasks.map(task => (
                <UserTaskCard key={task.id} task={task} onToggleStatus={toggleTaskStatus} />
              ))}
            </div>
          </div>
        )}

        {/* Completed Tasks */}
        {completedTasks.length > 0 && (
          <div className="bg-white rounded-xl shadow-sm border border-green-200">
            <div className="bg-green-50 px-6 py-4 border-b border-green-100 flex items-center gap-2">
              <Check className="w-5 h-5 text-green-600" />
              <h3 className="font-semibold text-green-900">Completed Tasks ({completedTasks.length})</h3>
            </div>
            <div className="divide-y divide-gray-100">
              {completedTasks.map(task => (
                <UserTaskCard key={task.id} task={task} onToggleStatus={toggleTaskStatus} />
              ))}
            </div>
          </div>
        )}

        {/* No Tasks */}
        {userTasks.length === 0 && (
          <div className="text-center py-12 bg-white rounded-xl border border-gray-100">
            <Users className="w-12 h-12 mx-auto text-gray-300 mb-4" />
            <p className="text-gray-500">No tasks assigned to you yet</p>
          </div>
        )}
      </div>
    );
  };

  return (
    <div className="min-h-screen bg-gray-50">
      {/* Header */}
      <div className="bg-white border-b border-gray-200 sticky top-0 z-10">
        <div className="max-w-7xl mx-auto px-6 py-4">
          <div className="flex items-center justify-between">
            <h1 className="text-2xl font-bold text-gray-900">Task Management</h1>
            <div className="flex items-center gap-3">
              {/* Quick Actions */}
              <button
                onClick={() => setShowTaskModal(true)}
                className="flex items-center gap-2 px-6 py-2.5 bg-blue-600 text-white rounded-lg hover:bg-blue-700 transition-colors font-medium"
              >
                <Plus className="w-5 h-5" />
                Add Task
              </button>
              
              {/* Advanced Menu Toggle */}
              <button
                onClick={() => setShowAdvancedMenu(!showAdvancedMenu)}
                className="p-2.5 text-gray-600 hover:bg-gray-100 rounded-lg transition-colors"
              >
                <MoreVertical className="w-5 h-5" />
              </button>
            </div>
          </div>
          
          {/* Advanced Menu Dropdown */}
          {showAdvancedMenu && (
            <div className="mt-3 flex gap-2 pb-2 border-t pt-3">
              {['my tasks', 'dashboard', 'timeline', 'associates'].map(view => (
                <button
                  key={view}
                  onClick={() => {
                    setCurrentView(view);
                    setShowAdvancedMenu(false);
                  }}
                  className={`px-4 py-2 rounded-lg text-sm font-medium transition-colors ${
                    currentView === view 
                      ? 'bg-blue-600 text-white' 
                      : 'text-gray-600 hover:bg-gray-100'
                  }`}
                >
                  {view === 'my tasks' ? 'My Tasks' : view.charAt(0).toUpperCase() + view.slice(1)}
                </button>
              ))}
            </div>
          )}
        </div>
      </div>

      {/* Main Content */}
      <div className="max-w-7xl mx-auto px-6 py-8">
        {currentView === 'dashboard' && <DashboardView />}
        {currentView === 'timeline' && <TimelineView />}
        {currentView === 'associates' && <AssociateView />}
        {currentView === 'my tasks' && <UserView />}
      </div>

      {/* Task Modal */}
      {showTaskModal && (
        <div className="fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center z-50 p-4">
          <div className="bg-white rounded-2xl max-w-2xl w-full max-h-[90vh] overflow-y-auto">
            <div className="sticky top-0 bg-white border-b border-gray-100 px-6 py-4 flex justify-between items-center">
              <h2 className="text-xl font-semibold">
                {editingTask ? 'Edit Task' : 'Create New Task'}
              </h2>
              <button onClick={() => { setShowTaskModal(false); resetForm(); }} className="text-gray-400 hover:text-gray-600">
                <X className="w-6 h-6" />
              </button>
            </div>
            <div className="p-6 space-y-4">
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-2">Project</label>
                <select
                  value={formData.project}
                  onChange={(e) => setFormData({...formData, project: e.target.value})}
                  className="w-full px-4 py-2 border border-gray-200 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
                  required
                >
                  <option value="">Select Project</option>
                  {PROJECTS.map(p => <option key={p} value={p}>{p}</option>)}
                </select>
              </div>

              <div>
                <label className="block text-sm font-medium text-gray-700 mb-2">Task Title</label>
                <div className="flex gap-2">
                  <input
                    type="text"
                    value={formData.title}
                    onChange={(e) => setFormData({...formData, title: e.target.value})}
                    className="flex-1 px-4 py-2 border border-gray-200 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
                    required
                  />
                  <button
                    type="button"
                    onClick={voiceInput}
                    className={`px-4 py-2 rounded-lg transition-colors ${isRecording ? 'bg-red-500 text-white' : 'bg-gray-100 text-gray-600 hover:bg-gray-200'}`}
                  >
                    <Mic className="w-5 h-5" />
                  </button>
                </div>
              </div>

              <div>
                <label className="block text-sm font-medium text-gray-700 mb-2">Description</label>
                <textarea
                  value={formData.description}
                  onChange={(e) => setFormData({...formData, description: e.target.value})}
                  className="w-full px-4 py-2 border border-gray-200 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
                  rows="3"
                />
              </div>

              <div className="grid grid-cols-2 gap-4">
                <div>
                  <label className="block text-sm font-medium text-gray-700 mb-2">Priority</label>
                  <select
                    value={formData.priority}
                    onChange={(e) => setFormData({...formData, priority: e.target.value})}
                    className="w-full px-4 py-2 border border-gray-200 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
                  >
                    <option value="Low">Low</option>
                    <option value="Medium">Medium</option>
                    <option value="High">High</option>
                  </select>
                </div>

                <div>
                  <label className="block text-sm font-medium text-gray-700 mb-2">Severity</label>
                  <select
                    value={formData.severity}
                    onChange={(e) => setFormData({...formData, severity: e.target.value})}
                    className="w-full px-4 py-2 border border-gray-200 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
                  >
                    <option value="Minor">Minor</option>
                    <option value="Major">Major</option>

                </button>

            </div>
          </div>

const UserTaskCard = ({ task, onToggleStatus }) => {
  const [expanded, setExpanded] = useState(false);
  const isOverdue = new Date(task.outDate) < new Date() && task.status !== 'Completed';
  const daysUntilDue = Math.ceil((new Date(task.outDate) - new Date()) / (1000 * 60 * 60 * 24));
  
  return (
    <div className="p-4 hover:bg-gray-50 transition-colors">
      <div className="flex items-start gap-4">
        <button
          onClick={() => onToggleStatus(task.id)}
          className={`mt-1 w-6 h-6 rounded-full border-2 flex items-center justify-center transition-colors ${
            task.status === 'Completed'
              ? 'bg-green-500 border-green-500'
              : 'border-gray-300 hover:border-blue-500'
          }`}
        >
          {task.status === 'Completed' && <Check className="w-4 h-4 text-white" />}
        </button>
        
        <div className="flex-1">
          <div className="flex items-center gap-3 mb-2">
            <h4 className={`font-medium ${task.status === 'Completed' ? 'text-gray-400 line-through' : 'text-gray-900'}`}>
              {task.title}
            </h4>
            <span className={`px-2 py-1 rounded text-xs font-medium ${PRIORITY_COLORS[task.priority]} border`}>
              {task.priority}
            </span>
            <span className={`px-2 py-1 rounded text-xs font-medium ${SEVERITY_BADGES[task.severity]}`}>
              {task.severity}
            </span>
          </div>
          
          <div className="flex items-center gap-4 text-sm text-gray-600 mb-2">
            <span className="flex items-center gap-1">
              <Calendar className="w-4 h-4" />
              Due: {new Date(task.outDate).toLocaleDateString()}
              {daysUntilDue >= 0 && task.status !== 'Completed' && (
                <span className={`ml-1 ${daysUntilDue <= 2 ? 'text-red-600 font-medium' : ''}`}>
                  ({daysUntilDue} {daysUntilDue === 1 ? 'day' : 'days'} left)
                </span>
              )}
            </span>
            <span className="flex items-center gap-1">
              <Users className="w-4 h-4" />
              {task.team}
            </span>
            {task.whatsapp && (
              <MessageCircle className="w-4 h-4 text-green-600" title="WhatsApp notification sent" />
            )}
          </div>

          {task.reminder && (
            <div className="flex items-center gap-1 text-sm text-blue-600 mb-2">
              <Bell className="w-4 h-4" />
              Reminder: {new Date(task.reminder).toLocaleString()}
            </div>
          )}

          <div className="text-sm text-gray-600">
            <span className="font-medium">Project:</span> {task.project}
          </div>

          {expanded && task.description && (
            <div className="mt-3 p-3 bg-gray-50 rounded-lg text-sm text-gray-700">
              {task.description}
            </div>
          )}

          {task.description && (
            <button
              onClick={() => setExpanded(!expanded)}
              className="mt-2 text-xs text-blue-600 hover:text-blue-700 font-medium"
            >
              {expanded ? 'Show less' : 'Show more'}
            </button>
          )}
        </div>
      </div>
    </div>
  );
};
